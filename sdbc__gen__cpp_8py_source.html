1"""Generate C/C++ headers and definition files from an SDBC file"""

3import sdbc_parse as sdbc
4
5SDBC_LOC = "SDBC/Hawkrod.sdbc"
6GEN_DEF_LOC = "libraries/SDBC/SDBC.def"
7GEN_FORM_LOC = "libraries/SDBC/Format.h"
8
9START_DOC = """/**
10 * AUTO GENERATED FILE - DO NOT EDIT
11 *
12 * SDBC Values, defined in C as their respective ID
13 */
14
15#include <stdint.h>
16
17"""
18VALUE_LINE = "#define {} {}\n"
19X_MAC = " X({}) \\\n"
20
21TYPE_CONV = { # Appoximate type by bit count
22 True: { # Signed
23 sdbc.NumType.INTEGER: {
24 1: 'bool',
25 8: 'int8_t',
26 16: 'int16_t',
27 32: 'int32_t',
28 64: 'int64_t',
29 },
30 sdbc.NumType.FLOATING: {
31 32: 'float',
32 64: 'double',
33 },
34 },
35 False: { # Unsigned
36 sdbc.NumType.INTEGER: {
37 1: 'bool',
38 8: 'uint8_t',
39 16: 'uint16_t',
40 32: 'uint32_t',
41 64: 'uint64_t',
42 },
43 sdbc.NumType.FLOATING: {
44 32: 'float',
45 64: 'double',
46 },
47 }
48}
49
50db = sdbc.parse_file(SDBC_LOC)
51
52
53def is_int(num):
54 return int(num) == num
55
56
57def main() -> None:
58 with open(GEN_DEF_LOC, "w", encoding="utf-8") as f:
59 f.write(START_DOC)
60 f.write(f"#define CAN_MESSAGE_NO {len(db.messages)}\n")
61
62 msg_x = "\n"
63
64 for msg in db.messages:
65 f.write("\n// " + msg.name + "\n")
66 for sig in msg.signals:
67
68 flow = f"| {msg.sender if msg.sender else ''} >> {sig.receiver if sig.receiver else ''}" if msg.sender or sig.receiver else None
69 desc = sig.description
70 comment = f" // {f'{desc}' if desc else ''}{' 'if desc and flow else ''}{flow if flow else ''}" if desc or flow else ''
71
72 f.write(VALUE_LINE.format(
73 f"POST_{sig.name}", f"{msg.uid}, {sig.bits}, {sig.position}{comment}"))
74 typ = sig.form.f_type.name.lower()
75 typ = TYPE_CONV[typ] if typ in TYPE_CONV else typ
76 if not sig.signed and typ.startswith("int"):
77 typ = "u" + typ
78
79 scale = "" if sig.scale == 1 else f"{sig.scale}"
80 offset = "" if sig.offset == 0 else f"{sig.offset}"
81
82 sver = "_s" if scale or offset else ""
83
84 scale = scale if scale else "1"
85 offset = offset if offset else "0"
86
87 calc_type = None
88
89 if is_int(sig.scale) and is_int(sig.offset):
90 calc_type = "uint32_t" if sig.bits <= 32 else "uint64_t"
91 else:
92 calc_type = "float" if sig.bits <= 32 else "double"
93
94 f.write(
95 VALUE_LINE.format(
96 f"POST_{sig.name}_META", f"{typ}, {32 if sig.bits <= 32 else 64}, {scale}, {offset}, {calc_type}, {sver}")
97 )
98 if msg is db.messages[-1]:
99 msg_x += X_MAC[:-2].format('"' + msg.name + '", ' + str(msg.can_id))
100 msg_x += "\n"
101 else:
102 msg_x += X_MAC.format('"' + msg.name + '", ' + str(msg.can_id))
103
104 f.write("\n#define CAN_IDS \\")
105 f.write(msg_x)
106
107
108if __name__ == "__main__":
109 main()
